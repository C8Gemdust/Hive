JavaScript (game.js) javascript Copy code const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const GRAVITY = 0.8; const FRICTION = 0.9; const PLAYER_SPEED = 5; const JUMP_STRENGTH = 15; let keys = {}; let level = 0; const levels = [ // Level 1 [ { x: 100, y: 300, width: 200, height: 20 }, { x: 400, y: 250, width: 200, height: 20 }, { x: 650, y: 200, width: 100, height: 20 }, { x: 750, y: 350, width: 50, height: 50 } ], // Level 2 [ { x: 50, y: 350, width: 150, height: 20 }, { x: 300, y: 300, width: 150, height: 20 }, { x: 550, y: 250, width: 150, height: 20 }, { x: 750, y: 200, width: 50, height: 50 } ], // Level 3 [ { x: 50, y: 350, width: 150, height: 20 }, { x: 250, y: 300, width: 100, height: 20 }, { x: 400, y: 250, width: 100, height: 20 }, { x: 550, y: 200, width: 100, height: 20 }, { x: 750, y: 150, width: 50, height: 50 } ] ]; const player = { x: 50, y: 50, width: 20, height: 20, speedX: 0, speedY: 0, jumping: false, }; function drawRect(obj) { ctx.fillStyle = '#000'; ctx.fillRect(obj.x, obj.y, obj.width, obj.height); } function drawPlayer() { ctx.fillStyle = '#FF0000'; ctx.fillRect(player.x, player.y, player.width, player.height); } function updatePlayer() { player.speedY += GRAVITY; player.speedX *= FRICTION; player.x += player.speedX; player.y += player.speedY; if (player.x + player.width > canvas.width) { player.x = canvas.width - player.width; } else if (player.x < 0) { player.x = 0; } if (player.y + player.height > canvas.height) { player.y = canvas.height - player.height; player.jumping = false; } levels[level].forEach(platform => { if (player.x < platform.x + platform.width && player.x + player.width > platform.x && player.y < platform.y + platform.height && player.y + player.height > platform.y) { player.speedY = 0; player.y = platform.y - player.height; player.jumping = false; } }); if (player.x + player.width > levels[level][levels[level].length - 1].x && player.y + player.height > levels[level][levels[level].length - 1].y) { level++; if (level >= levels.length) { alert('You win!'); level = 0; } player.x = 50; player.y = 50; player.speedX = 0; player.speedY = 0; player.jumping = false; } } function drawLevel() { levels[level].forEach(platform => drawRect(platform)); } function gameLoop() { ctx.clearRect(0, 0, canvas.width, canvas.height); updatePlayer(); drawLevel(); drawPlayer(); requestAnimationFrame(gameLoop); } window.addEventListener('keydown', e => { keys[e.key] = true; if ((keys['ArrowUp'] || keys['w']) && !player.jumping) { player.jumping = true; player.speedY = -JUMP_STRENGTH; } if (keys['ArrowRight'] || keys['d']) { player.speedX = PLAYER_SPEED; } if (keys['ArrowLeft'] || keys['a']) { player.speedX = -PLAYER_SPEED; } }); window.addEventListener('keyup', e => { keys[e.key] = false; if (e.key === 'ArrowRight' || e.key === 'd') { player.speedX = 0; } if (e.key === 'ArrowLeft' || e.key === 'a') { player.speedX = 0; } }); gameLoop();
